import streamlit as st
import google.generativeai as genai
import os
import pandas as pd

# --- 設定 ---
st.set_page_config(page_title="Wire-Guard AI", layout="wide")
st.title("🏗️ ワイヤー点検AI解析システム")

# サイドバー設定
api_key = st.sidebar.text_input("Google AI Studio API Keyを入力", type="password")

# --- アプリのロジック ---
if api_key:
    genai.configure(api_key=api_key)
    model = genai.GenerativeModel("gemini-1.5-flash")

    machine_id = st.sidebar.selectbox("杭打機を選択",[f"機体-{i:03}" for i in range(1, 31)])
    uploaded_file = st.sidebar.file_uploader("点検動画をアップロード", type=["mp4", "mov"])

    if uploaded_file:
        st.video(uploaded_file)
        if st.button("AI解析を実行"):
            with st.spinner("AIが映像を解析中です..."):
                # ここで動画データをGeminiに渡す処理を実装
                # ※実際はファイルアップロード後に一時保存したパスを渡します
                st.success("解析完了")
                st.subheader("解析結果レポート")
                
                # サンプル出力
                results = pd.DataFrame({
                    "時刻": ["00:10", "01:20"],
                    "事象": ["摩耗", "素線切れ"],
                    "重症度": ["注意", "警告"]
                })
                st.table(results)
                
                # 修正機能UI
                st.subheader("AI判定の修正・フィードバック")
                correction = st.text_area("AIの判定が誤っている場合、詳細を入力してください")
                if st.button("修正を送信"):
                    st.success("フィードバックを記録しました。精度改善に活用します。")
else:
    st.info("左側のサイドバーにAPIキーを入力してください。")
